list = ["data_quiz/wdi_agedist_1980.csv", "data_quiz/wdi_agedist_2000.csv", "data_quiz/wdi_agedist_2020.csv"]
index = 0
for i in list:
    dataset = pd.read_csv(i)
    percent_ages0to14 = dataset["percent_ages0to14"]
    proportion = 100 - percent_ages0to14
    plt.hist(x = proportion)
    plt.xlabel("proportion of population 15 or older")
    plt.ylabel("frequency")
    plt.show()
    index = index+1
